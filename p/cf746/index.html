const int M = 1e5 + 7;

struct Node {
    int to, nxt;
} tr[M * 2];

int head[M], idx, num[M], xxor[M], all, cnt;

void add(int u, int v) {
    tr[idx] = {v, head[u]};
    head[u] = idx ++;
}

int dfs(int u, int fa) {
    int now = num[u];
    for(int i = head[u]; ~i; i = tr[i].nxt) {
        int j = tr[i].to;
        if(j != fa) {
            now ^= dfs(j, u);
        }
    }
    if(now == all) {
        cnt ++;
        return 0;
    }
    return xxor[u] = now;
}

int main() {
    int t = read();
    while(t --) {
        idx = 0, all = 0, cnt = 0;
        ll n = read(), k = read();
        for(int i = 1; i <= n; i ++) {
            num[i] = read();
            head[i] = -1;
            all ^= num[i];
        }

        for(int i = 1; i < n ; i ++) {
            int u = read(), v = read();
            add(u, v);
            add(v, u);
        }
        dfs(1, - 1);
        if(all == 0) {
            cout << "YES" << endl;
        } else {
            dfs(1, -1);
            if(k >= 3 && cnt > 2) cout << "YES" << endl;
            else cout << "NO" << endl;
        }
    }
    return 0;
}
